220 void HaplotypeScorer::find_possible_haplotypes(std::vector<std::vector<sgNodeID_t >> bubbles){
221  // algorithm: https://stackoverflow.com/questions/1867030/combinations-of-multiple-vectors-elements-without-repetition
222  size_t P = 1;
223  auto N =bubbles.size();
224  for(size_t i=0;i<N;++i) {
225  P *= bubbles[i].size();
226  haplotype_nodes.insert(bubbles[i].begin(), bubbles[i].end());
227  }
228  std::vector<std::vector<sgNodeID_t >> haps;
229  std::cout << P << " combinations to generate from " << N << " bubbles " << std::endl;
230  for (size_t m=0; m < P; m++ ) {
231  // this should hold the index to take from each bubble
232  std::vector<size_t> indices(N);
233  std::vector<sgNodeID_t > bubble;
234  size_t m_curr = m;
235  for (size_t i = 0; i < N; ++i) {
236  indices[i] = m_curr% bubbles[i].size();
237  bubble.push_back(bubbles[i][indices[i]]);
238  m_curr /= bubbles[i].size();
239  }
240 
241  haplotype_ids.push_back(bubble);
242  }
243  std::cout << haplotype_ids.size() << " haplotypes generated " << std::endl;
